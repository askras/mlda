–û–ø—Ä–µ–¥–µ–ª–∏–º —Ñ—É–Ω–∫—Ü–∏–æ–Ω–∞–ª –∫–∞—á–µ—Å—Ç–≤–∞ –∞–ø–ø—Ä–æ–∫—Å–∏–º–∞—Ü–∏–∏ —Ü–µ–ª–µ–≤–æ–π –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –Ω–∞ –≤—ã–±–æ—Ä–∫–µ $ùëã^‚Ñì$ –∫–∞–∫ —Å—É–º–º—É –∫–≤–∞–¥—Ä–∞—Ç–æ–≤ –æ—à–∏–±–æ–∫

\begin{equation}
\label{1}\tag{1}
Q(\alpha, X^l) = \sum\limits_{i=1}^{l}(g(x_i, \alpha) - y_i)^2.
\end{equation}

–û–±—É—á–µ–Ω–∏–µ –ø–æ –º–µ—Ç–æ–¥—É –Ω–∞–∏–º–µ–Ω—å—à–∏—Ö –∫–≤–∞–¥—Ä–∞—Ç–æ–≤ (–ú–ù–ö) —Å–æ—Å—Ç–æ–∏—Ç –≤ —Ç–æ–º, —á—Ç–æ–±—ã –Ω–∞–π—Ç–∏ –≤–µ–∫—Ç–æ—Ä –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ ùõº‚àó, –ø—Ä–∏ –∫–æ—Ç–æ—Ä–æ–º –¥–æ—Å—Ç–∏–≥–∞–µ—Ç—Å—è –º–∏–Ω–∏–º—É–º —Å—Ä–µ–¥–Ω–µ–≥–æ

\begin{equation}
\label{2}\tag{2}
\alpha^{*} = \arg\min\limits_{\alpha \in  ‚Ñù^p} Q(\alpha, X^l).
\end{equation}
